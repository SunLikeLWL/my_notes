```js
const isValid = (s)=>{
    while(true){
        let len = s.length;
        s = s.replace('{}','').replace('[]','').replace('()','')
        // 有两种情况s.length会等于len
        // s匹配完成了，编程了空字符串
        // 是无法继续匹配，导致其长度和一开始的len一样
        if(s.length=== len){
            return len === 0
        }
    }
}
```

```js


const isValid = (s)=>{
    if(!s){
        return true
    }
    const leftToRight = {
        '(':')',
        '[':']',
        '{':'}'
    }
   
   const stack = []

   for(let i=0;,len = s.length;i<len;i++){
       const ch = s[i];
       if(leftToRight[ch]){
           stack.push(ch)
       }else{
           // 右侧号开始匹配
           // 1、如果栈内没有左括号，直接false
           // 2、有数据但是栈顶元素不是当前的右括号
           if(!stack.length || leftToRight[stack.pop()]!==ch){
               return false
           }
       }
   }
   
   return !stack.length
}
```
